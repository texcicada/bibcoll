\newbibmacro*{artist}{%
  \printnames{artist}%
  \setunit{\addcomma\addspace}%
}
  
\newbibmacro*{artworkname}{%
  \printfield{title}%
  \setunit{\addcomma\addspace}%
  \printfield{year}%
  }

	\addcolon\addspace

}% end bibmacro
